class WordDictionary:

    def __init__(self):
        self.store = []

    def addWord(self, word: str) -> None:
        self.store.append(word)

    def search(self, word: str) -> bool:
        for w in self.store:
            if len(w) != len(word):
                continue
            i = 0
            while i < len(w):
                if w[i] == word[i] or word[i] == '.':
                    i += 1
                else:
                    break
            if i == len(w):
                return True
        return False

Time & Space Complexity

  • Time complexity: O(1)O(1)O(1) for addWord()addWord()addWord(), O(m∗n)O(m * n)O(m∗n) for search()search()search().
  • Space complexity: O(m∗n)O(m * n)O(m∗n)

Where mmm is the number of words added and nnn is the length of the string.

class TrieNode:
    def __init__(self):
        self.children = {}
        self.word = False


class WordDictionary:
    def __init__(self):
        self.root = TrieNode()

    def addWord(self, word: str) -> None:
        cur = self.root
        for c in word:
            if c not in cur.children:
                cur.children[c] = TrieNode()
            cur = cur.children[c]
        cur.word = True

    def search(self, word: str) -> bool:
        def dfs(j, root):
            cur = root

            for i in range(j, len(word)):
                c = word[i]
                if c == ".":
                    for child in cur.children.values():
                        if dfs(i + 1, child):
                            return True
                    return False
                else:
                    if c not in cur.children:
                        return False
                    cur = cur.children[c]
            return cur.word

        return dfs(0, self.root)

Time & Space Complexity

  • Time complexity: O(n)O(n)O(n) for addWord()addWord()addWord(), O(n)O(n)O(n) for search()search()search().
  • Space complexity: O(t+n)O(t + n)O(t+n)

Where nnn is the length of the string and ttt is the total number of TrieNodes created in the Trie.
